### 运算符
| 操作符 | 方法                    |
| :----- | :---------------------- |
| +      | a.plus(b)               |
| -      | a.minus(b)              |
| *      | a.multiply(b)           |
| /      | a.div(b)                |
| %      | a.mod(b)                |
| **     | a.power(b)              |
| \|     | a.or(b)                 |
| &      | a.and(b)                |
| ^      | a.xor(b)                |
| as     | a.asType(b)             |
| a(b)   | a.call(b)               |
| a[b]   | a.getAt(b)              |
| a[b]=c | a.putAt(b, c)           |
| a in b | b.isCase(a)             |
| <<     | a.leftShift(b)          |
| >>     | a.rightShift(b)         |
| >>>    | a.rightShiftUnsigned(b) |
| ++     | a.next(b)               |
| --     | a.previous(b)           |
| +a     | a.positive(b)           |
| -a     | a.negative(b)           |
| ~a     | a.bitwiseNegate(b)      |
